// Portfolio Calculator Module
const Portfolio = {
  data: null,
  
  init(cryptoData) {
    this.data = cryptoData;
    this.render();
    this.startLiveUpdates();
  },
  
  calculatePortfolioValue() {
    return this.data.holdings.reduce((total, crypto) => {
      return total + (crypto.amount * crypto.currentPrice);
    }, 0);
  },
  
  calculateTotalCost() {
    return this.data.holdings.reduce((total, crypto) => {
      return total + (crypto.amount * crypto.purchasePrice);
    }, 0);
  },
  
  calculateProfitLoss() {
    const currentValue = this.calculatePortfolioValue();
    const totalCost = this.calculateTotalCost();
    return {
      amount: currentValue - totalCost,
      percentage: ((currentValue - totalCost) / totalCost) * 100
    };
  },
  
  formatCurrency(value) {
    return new Intl.NumberFormat('en-US', {
      style: 'currency',
      currency: 'USD',
      minimumFractionDigits: 2,
      maximumFractionDigits: 2
    }).format(value);
  },
  
  render() {
    // Update portfolio summary
    const totalValue = this.calculatePortfolioValue();
    const profitLoss = this.calculateProfitLoss();
    
    document.getElementById('totalValue').textContent = this.formatCurrency(totalValue);
    
    const changeEl = document.getElementById('totalChange');
    const isPositive = profitLoss.amount >= 0;
    changeEl.innerHTML = `
      <span class="change-icon">${isPositive ? '↑' : '↓'}</span>
      <span class="change-percent">${Math.abs(profitLoss.percentage).toFixed(2)}%</span>
      <span class="change-amount">(${this.formatCurrency(Math.abs(profitLoss.amount))})</span>
    `;
    changeEl.className = `summary-change ${isPositive ? 'positive' : 'negative'}`;
    
    // Render crypto cards
    const grid = document.getElementById('cryptoGrid');
    grid.innerHTML = this.data.holdings.map(crypto => this.renderCard(crypto)).join('');
    
    // Update timestamp
    document.getElementById('lastUpdate').textContent = new Date().toLocaleTimeString();
  },
  
  renderCard(crypto) {
    const value = crypto.amount * crypto.currentPrice;
    const cost = crypto.amount * crypto.purchasePrice;
    const profit = value - cost;
    const isPositive24h = crypto.priceChange24h >= 0;
    const isProfitable = profit >= 0;
    
    return `
      <article class="crypto-card">
        <div class="card-header">
          <span class="crypto-icon">${crypto.icon}</span>
          <div class="crypto-name">
            <h3>${crypto.name}</h3>
            <span class="symbol">${crypto.symbol}</span>
          </div>
        </div>
        <div class="card-body">
          <div class="price-row">
            <span class="label">Price</span>
            <span class="value">${this.formatCurrency(crypto.currentPrice)}</span>
          </div>
          <div class="price-row">
            <span class="label">24h</span>
            <span class="value ${isPositive24h ? 'positive' : 'negative'}">
              ${isPositive24h ? '+' : ''}${crypto.priceChange24h.toFixed(2)}%
            </span>
          </div>
          <div class="price-row">
            <span class="label">Holdings</span>
            <span class="value">${crypto.amount} ${crypto.symbol}</span>
          </div>
          <div class="price-row total">
            <span class="label">Value</span>
            <span class="value">${this.formatCurrency(value)}</span>
          </div>
          <div class="price-row profit">
            <span class="label">P/L</span>
            <span class="value ${isProfitable ? 'positive' : 'negative'}">
              ${isProfitable ? '+' : ''}${this.formatCurrency(profit)}
            </span>
          </div>
        </div>
      </article>
    `;
  },
  
  startLiveUpdates() {
    setInterval(() => {
      // Simulate price fluctuations
      this.data.holdings.forEach(crypto => {
        const fluctuation = (Math.random() - 0.5) * 0.02;
        crypto.currentPrice *= (1 + fluctuation);
        crypto.priceChange24h += (Math.random() - 0.5) * 0.1;
      });
      this.render();
    }, 3000);
  }
};
